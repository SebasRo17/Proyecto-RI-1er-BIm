I have to show with arrow symbols like on this picture: ![the correct painting](http://i.stack.imgur.com/9Wvd7.png) The closest to the picture is this plot with lane curves. Clear[x, y, xdot, ydot, curves, zerocurves, specielarrows]     xdot = x^2 + y^2 - 25;     ydot = x^2 - y^2 - 7;     xmin = -6;     xmax = 6;     ymin = -6;     ymax = 6;     curves = 30;     specielarrows = {{{6, 0}, Directive[Orange, Thick]}};     {Append[specielarrows, curves]};          (* plots *)     zerocurves =       ContourPlot[{xdot == 0, ydot == 0}, {x, xmin, xmax}, {y, ymin, ymax},        ContourStyle -> {Directive[Red, Thick], Directive[Blue, Thick]},        Frame -> False, Axes -> True, AxesLabel -> {x, y}]                 lanecurves =       VectorPlot[{xdot, ydot}, {x, xmin, xmax}, {y, ymin, ymax},        StreamPoints -> {Append[specielarrows, curves]},        StreamStyle -> Green, VectorStyle -> Black, Axes -> True,        Frame -> False, AxesLabel -> {x, y}]          Show[zerocurves, lanecurves]
